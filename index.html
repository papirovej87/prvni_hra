  <script>

    let stage = 0;


    function nextStage(text, choices) {

      document.getElementById('story').innerText = text;

      const choiceDiv = document.getElementById('choices');

      choiceDiv.innerHTML = '';

      choices.forEach(c => {

        const btn = document.createElement('button');

        btn.textContent = c.text;

        btn.onclick = c.func;

        choiceDiv.appendChild(btn);

      });

    }


    function choice1() {

      nextStage(

        "Na hřišti si tě všimne trenér Sparty. Co uděláš?",

        [

          { text: "Ukážeš kličku!", func: choice2a },

          { text: "Schováš se za maminku.", func: gameOver1 }

        ]

      );

    }


    function choice2a() {

      nextStage(

        "Trenér je nadšený! Pozve tě na trénink do Sparty. Tam se tě kluci ptají: Umíš dát gól patičkou?",

        [

          { text: "Jasně, sledujte!", func: choice3a },

          { text: "Ne, nikdy jsem to nezkoušel.", func: choice3b }

        ]

      );

    }


    function gameOver1() {

      nextStage(

        "Schoval ses, trenér tě neviděl. Sen o Spartě se rozplynul...",

        [

          { text: "Zkusit znovu", func: () => location.reload() }

        ]

      );

    }


    function choice3a() {

      nextStage(

        "Gól patičkou! Všichni tleskají. O pár let později tě čeká zápas o smlouvu se Spartou. Máš horečku. Co uděláš?",

        [

          { text: "Hraješ i s horečkou.", func: finale },

          { text: "Zůstaneš doma.", func: gameOver2 }

        ]

      );

    }


    function choice3b() {

      nextStage(

        "Kluci se ti smějí, ale nevzdáš to. O pár let později tě čeká zápas o smlouvu se Spartou. Máš horečku. Co uděláš?",

        [

          { text: "Hraješ i s horečkou.", func: finale },

          { text: "Zůstaneš doma.", func: gameOver2 }

        ]

      );

    }


    function gameOver2() {

      nextStage(

        "Tým bez tebe prohrál. Sen o Spartě je pryč... ale zdraví je důležitější.",

        [

          { text: "Zkusit znovu", func: () => location.reload() }

        ]

      );

    }


    function finale() {

      nextStage(

        "⭐ Podepsal jsi smlouvu se Spartou! Letná zpívá tvé jméno: MAAA-TÝ-ŠEK! ⭐",

        [

          { text: "Hrát znovu", func: () => location.reload() }

        ]

      );

    }
